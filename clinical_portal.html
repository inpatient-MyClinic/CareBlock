""" CareBlock Fast Deploy Script Run this on any computer with Python 3 + internet access: python update_portal.py It uses the GitHub API to update files directly — no git needed. Set your token once and it remembers. """ import os, sys, base64, json, urllib.request, urllib.error, getpass REPO = "inpatient-MyClinic/CareBlock" BRANCH = "main" FILES = { "clinical_portal.html": "clinical_portal.html", "app.py": "app.py", } TOKEN_FILE = os.path.join(os.path.expanduser("~"), ".careblock_token") def get_token(): if os.path.exists(TOKEN_FILE): with open(TOKEN_FILE) as f: tok = f.read().strip() if tok: print(f"✅ Using saved GitHub token") return tok print("\n📋 GitHub Personal Access Token needed") print(" Get one at: https://github.com/settings/tokens/new") print(" Scope needed: ✅ repo (full control)") tok = getpass.getpass("Paste token: ").strip() with open(TOKEN_FILE, "w") as f: f.write(tok) os.chmod(TOKEN_FILE, 0o600) print(f"✅ Token saved to {TOKEN_FILE}") return tok def api(path, method="GET", body=None, token=None): url = f"https://api.github.com{path}" headers = { "Authorization": f"token {token}", "Accept": "application/vnd.github.v3+json", "User-Agent": "CareBlock-Deploy" } data = json.dumps(body).encode() if body else None req = urllib.request.Request(url, data=data, headers=headers, method=method) try: with urllib.request.urlopen(req) as r: return json.loads(r.read()) except urllib.error.HTTPError as e: print(f"❌ GitHub API error {e.code}: {e.read().decode()[:200]}") sys.exit(1) def deploy(local_dir="."): token = get_token() print(f"\n🚀 Deploying to {REPO} / {BRANCH}") for local_name, remote_name in FILES.items(): local_path = os.path.join(local_dir, local_name) if not os.path.exists(local_path): print(f"⚠️ {local_name} not found — skipping") continue with open(local_path, "rb") as f: content = f.read() size_kb = len(content) // 1024 print(f"\n📤 {local_name} ({size_kb}KB)...") # Get current file SHA (needed for update) existing = api(f"/repos/{REPO}/contents/{remote_name}?ref={BRANCH}", token=token) sha = existing.get("sha", "") # Upload body = { "message": f"Update {remote_name} via deploy script", "content": base64.b64encode(content).decode(), "branch": BRANCH, } if sha: body["sha"] = sha api(f"/repos/{REPO}/contents/{remote_name}", method="PUT", body=body, token=token) print(f" ✅ {remote_name} updated on GitHub") print(f"\n✅ Done! Railway will redeploy automatically in ~2 minutes.") print(f" Watch: https://railway.app/dashboard") print(f" Live: https://web-production-673769.up.railway.app") if __name__ == "__main__": # Look for files next to this script script_dir = os.path.dirname(os.path.abspath(__file__)) deploy(script_dir)
